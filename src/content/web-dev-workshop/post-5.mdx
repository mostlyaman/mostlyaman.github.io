#### 5. String Methods

```js
const alphabet = "aBcDe"
const length = alphabet.length              // 5
const uppercase = alphabet.toUpperCase()    // ABCDE
const lowercase = alphabet.toLowerCase()    // abcde

const characters = alphabet.split("")         // ["a", "B", "c", "D", "e"]

console.log(length, uppercase, lowercase, substring, characters)
```

#### 6. Array Methods

```js
let array = [1, 2, 3, 4, 5]
let array = new Array(1, 2, 3, 4, 5)

console.log(`First Element: ${array[0]}`)            // Indexing
console.log(`Index of Value 4: ${array.indexOf(4)}`) // Finding Index of Element
array.push(6)                                        // Adding a element at end
array.unshift(0)                                     // Adding a element at front
console.log(array.pop())                             // Remove last element
console.log(array.shift())                           // Remove first element
```

##### Iterating over a Array

<Remember text="To iterate over an array, you can either use the for loop, or the forEach method." />

```js
let array = [2, 3, 4]

for(let i=0; i<array.length; i++) {
  console.log(`Element ${i}: ${array[i]}`)
}

function print(x) {
  console.log(x)
}
array.forEach(print)
```

#### 7. Objects

<Remember text="Objects are used to store data in JSON format. An object can behave like a class, by storing methods which can access its own properties. " />

```js
let object = {
  
  // Literals
  firstName: "John",         
  lastName: "Doe",
  
  // Methods
  fullName: function () { return this.firstName + " " + this.lastName },
  
  // Nesting
  address: {
    street: "Street 2",
    city: "Mandi"
  }
}

// Object Method
console.log(object.fullName())

console.log(object.toString()) 

let string = JSON.stringify(object)
console.log(string)
```

#### ❗❗IMPORTANT❗❗
<Remember text="Objects are passed by reference (like pointers) by default." />

```js
let a = { key1: "value1", key2: "value2" }
let b = a
b.key1 = "value3"

// Both a and b gets updated because they point to the same reference
console.log("a becomes: " + JSON.stringify(a))
console.log("b becomes: " + JSON.stringify(b))
```
